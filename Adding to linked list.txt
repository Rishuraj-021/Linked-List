#include <iostream>
using namespace std;
class link{
    public:
    int data;
    link*next;
    link(int num){
        data=num;
        next=0;
    }
};
void insert_head(link*&head,int data){
    link*new_node=new link(data);
    new_node->next=head;
    head=new_node;
}
void disp(link*head){
    link*temp=head;
    while(temp!=0){
        cout<<temp->data<<"->";
        temp=temp->next;
    }
    cout<<"0"<<endl;
}
int main() {
    link*head=0;
    insert_head(head,20);
    disp(head);
    insert_head(head,50);
    disp(head);
    insert_head(head,80);
    disp(head);
    return 0;
}
